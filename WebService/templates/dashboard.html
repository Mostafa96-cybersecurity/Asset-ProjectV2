async function loadDashboardData() { try { const response = await fetch('/api/stats'); const data = await response.json(); updateDashboardStats(data); createCharts(data); } catch (error) { console.error('Error loading dashboard data:', error); showError('Failed to load dashboard data'); } } // Update dashboard statistics function updateDashboardStats(data) { // Quick stats banner document.getElementById('totalDevices').innerHTML = data.total_devices; document.getElementById('aliveDevices').innerHTML = data.device_status.alive || 0; document.getElementById('hardwareComplete').innerHTML = data.devices_with_processor; document.getElementById('hostnameMismatches').innerHTML = data.hostname_mismatches; // Stats cards document.getElementById('totalAssetsCard').innerHTML = data.total_devices; document.getElementById('processorDataCard').innerHTML = data.devices_with_processor; document.getElementById('memoryDataCard').innerHTML = data.devices_with_memory; document.getElementById('macAddressCard').innerHTML = data.devices_with_mac; document.getElementById('mismatchCard').innerHTML = data.hostname_mismatches; document.getElementById('recentUpdatesCard').innerHTML = data.recent_updates; // Update timestamp document.getElementById('lastUpdate').innerHTML = new Date().toLocaleString(); } // Create charts function createCharts(data) { // Device Status Chart const statusCtx = document.getElementById('statusChart').getContext('2d'); if (statusChart) statusChart.destroy(); const statusLabels = Object.keys(data.device_status); const statusValues = Object.values(data.device_status); const statusColors = statusLabels.map(status => { switch(status) { case 'alive': return '#059669'; case 'dead': return '#dc2626'; default: return '#64748b'; } }); statusChart = new Chart(statusCtx, { type: 'doughnut', data: { labels: statusLabels, datasets: [{ data: statusValues, backgroundColor: statusColors, borderWidth: 2, borderColor: '#fff' }] }, options: { responsive: true, plugins: { legend: { position: 'bottom' } } } }); // Operating System Chart const osCtx = document.getElementById('osChart').getContext('2d'); if (osChart) osChart.destroy(); const osLabels = data.operating_systems.map(os => os.name || 'Unknown'); const osValues = data.operating_systems.map(os => os.count); osChart = new Chart(osCtx, { type: 'bar', data: { labels: osLabels, datasets: [{ label: 'Device Count', data: osValues, backgroundColor: '#2563eb', borderColor: '#1d4ed8', borderWidth: 1 }] }, options: { responsive: true, scales: { y: { beginAtZero: true } }, plugins: { legend: { display: false } } } }); // Collection Methods Chart const collectionCtx = document.getElementById('collectionChart').getContext('2d'); if (collectionChart) collectionChart.destroy(); const collectionLabels = data.collection_methods.map(method => method.method || 'Unknown'); const collectionValues = data.collection_methods.map(method => method.count); collectionChart = new Chart(collectionCtx, { type: 'pie', data: { labels: collectionLabels, datasets: [{ data: collectionValues, backgroundColor: [ '#2563eb', '#059669', '#d97706', '#dc2626', '#0891b2', '#7c3aed' ], borderWidth: 2, borderColor: '#fff' }] }, options: { responsive: true, plugins: { legend: { position: 'bottom' } } } }); } // Refresh dashboard function refreshDashboard() { loadDashboardData(); } // Export data function exportData() { // This would trigger a download of CSV/Excel file alert('Export functionality coming soon!');
